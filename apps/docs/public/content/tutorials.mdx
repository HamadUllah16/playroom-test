## First Game Tutorial

Let's build your first multiplayer game - a simple counter that players can increment together.

### Step 1: Setup

Create a new HTML file:

```html
<!DOCTYPE html>
<html>
<head>
    <title>My First Multiplayer Game</title>
    <script src="https://cdn.PlayroomKit.com/client.js"></script>
</head>
<body>
    <div id="game">
        <h1>Multiplayer Counter</h1>
        <div id="counter">0</div>
        <button id="increment">Increment</button>
        <div id="players"></div>
    </div>
    <script src="game.js"></script>
</body>
</html>
```

### Step 2: Initialize PlayroomKit

Create `game.js`:

```javascript
class CounterGame {
    constructor() {
        this.counter = 0;
        this.players = [];
        this.init();
    }
    
    async init() {
        // Initialize PlayroomKit
        await Playroom.init('your-api-key-here');
        
        // Create or join a room
        const room = await Playroom.createRoom({
            maxPlayers: 4,
            roomCode: 'COUNTER'
        });
        
        console.log(`Room created: ${room.code}`);
        
        // Set up event listeners
        this.setupEvents();
        
        // Initial render
        this.render();
    }
    
    setupEvents() {
        // Listen for player events
        Playroom.onPlayerJoin((player) => {
            console.log(`${player.name} joined!`);
            this.updatePlayers();
        });
        
        Playroom.onPlayerLeave((player) => {
            console.log(`${player.name} left!`);
            this.updatePlayers();
        });
        
        // Listen for counter updates
        Playroom.onStateChange('counter', (newValue) => {
            this.counter = newValue;
            this.render();
        });
        
        // Handle increment button
        document.getElementById('increment').addEventListener('click', () => {
            this.incrementCounter();
        });
    }
    
    incrementCounter() {
        const newValue = (Playroom.getState('counter') || 0) + 1;
        Playroom.setState('counter', newValue);
    }
    
    updatePlayers() {
        this.players = Playroom.getPlayers();
        this.render();
    }
    
    render() {
        // Update counter display
        document.getElementById('counter').textContent = this.counter;
        
        // Update players list
        const playersDiv = document.getElementById('players');
        playersDiv.innerHTML = `
            <h3>Players (${this.players.length})</h3>
            ${this.players.map(player => 
                `<div>${player.name} ${player.isHost ? 'ðŸ‘‘' : ''}</div>`
            ).join('')}
        `;
    }
}

// Start the game
const game = new CounterGame();
```

### Step 3: Test Your Game

1. Replace `'your-api-key-here'` with your actual API key
2. Open the HTML file in multiple browser tabs
3. Click the increment button in different tabs
4. Watch the counter synchronize across all instances!

### What's Happening?

- **Playroom.init()** - Connects to PlayroomKit servers
- **Playroom.createRoom()** - Creates a multiplayer room
- **Playroom.setState()** - Updates shared game state
- **Playroom.onStateChange()** - Listens for state updates
- **Playroom.onPlayerJoin/Leave()** - Handles player events
