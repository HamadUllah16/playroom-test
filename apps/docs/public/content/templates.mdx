## Tic Tac Toe Template

### React Version

```jsx
import { usePlayroom, usePlayroomState, useRpc } from '@PlayroomKit/react';

function TicTacToe() {
    const { players, myPlayer } = usePlayroom();
    const [board, setBoard] = usePlayroomState('board', Array(9).fill(null));
    const [currentPlayer, setCurrentPlayer] = usePlayroomState('currentPlayer', 'X');
    const [winner, setWinner] = usePlayroomState('winner', null);
    
    const makeMove = useRpc('makeMove');
    
    useRpc('makeMove', (data) => {
        const newBoard = [...board];
        newBoard[data.position] = data.symbol;
        setBoard(newBoard);
        
        const gameWinner = checkWinner(newBoard);
        if (gameWinner) {
            setWinner(gameWinner);
        } else {
            setCurrentPlayer(currentPlayer === 'X' ? 'O' : 'X');
        }
    });
    
    const handleClick = (index) => {
        if (board[index] || winner || !isMyTurn()) return;
        makeMove({ position: index, symbol: getMySymbol() });
    };
    
    const isMyTurn = () => {
        return (currentPlayer === 'X' && myPlayer.id === players[0]?.id) ||
               (currentPlayer === 'O' && myPlayer.id === players[1]?.id);
    };
    
    const getMySymbol = () => {
        return myPlayer.id === players[0]?.id ? 'X' : 'O';
    };
    
    return (
        <div className="tic-tac-toe">
            <h2>Tic Tac Toe</h2>
            {winner ? (
                <p>{winner === 'tie' ? "It's a tie!" : `Player ${winner} wins!`}</p>
            ) : (
                <p>Current turn: {currentPlayer}</p>
            )}
            
            <div className="board">
                {board.map((cell, index) => (
                    <button
                        key={index}
                        className="cell"
                        onClick={() => handleClick(index)}
                        disabled={!isMyTurn() || cell !== null || winner}
                    >
                        {cell}
                    </button>
                ))}
            </div>
            
            {winner && (
                <button onClick={() => resetGame()}>
                    Play Again
                </button>
            )}
        </div>
    );
}

function checkWinner(board) {
    const lines = [
        [0, 1, 2], [3, 4, 5], [6, 7, 8], // Rows
        [0, 3, 6], [1, 4, 7], [2, 5, 8], // Columns
        [0, 4, 8], [2, 4, 6] // Diagonals
    ];
    
    for (let [a, b, c] of lines) {
        if (board[a] && board[a] === board[b] && board[a] === board[c]) {
            return board[a];
        }
    }
    
    return board.every(cell => cell !== null) ? 'tie' : null;
}
```

### JavaScript Version

```javascript
import { Playroom } from '@PlayroomKit/client';

class TicTacToe {
    constructor() {
        this.board = Array(9).fill(null);
        this.currentPlayer = 'X';
        this.players = [];
        this.winner = null;
        
        this.init();
    }
    
    async init() {
        await Playroom.init('your-api-key');
        
        // Create or join room
        const room = await Playroom.createRoom({ maxPlayers: 2 });
        
        // Set up event listeners
        Playroom.onPlayerJoin(this.onPlayerJoin.bind(this));
        Playroom.onRpc('makeMove', this.onMakeMove.bind(this));
        Playroom.onStateChange('board', this.onBoardChange.bind(this));
        
        this.render();
    }
    
    onPlayerJoin(player) {
        this.players = Playroom.getPlayers();
        this.render();
    }
    
    onMakeMove(data, sender) {
        if (this.board[data.position] || this.winner) return;
        
        this.board[data.position] = data.symbol;
        Playroom.setState('board', this.board);
        
        const winner = this.checkWinner();
        if (winner) {
            this.winner = winner;
            Playroom.setState('winner', winner);
        } else {
            this.currentPlayer = this.currentPlayer === 'X' ? 'O' : 'X';
            Playroom.setState('currentPlayer', this.currentPlayer);
        }
    }
    
    makeMove(position) {
        if (!this.isMyTurn() || this.board[position] || this.winner) return;
        
        Playroom.rpc('makeMove', {
            position,
            symbol: this.getMySymbol()
        });
    }
    
    isMyTurn() {
        const myPlayer = Playroom.getMyPlayer();
        return (this.currentPlayer === 'X' && myPlayer.id === this.players[0]?.id) ||
               (this.currentPlayer === 'O' && myPlayer.id === this.players[1]?.id);
    }
    
    getMySymbol() {
        const myPlayer = Playroom.getMyPlayer();
        return myPlayer.id === this.players[0]?.id ? 'X' : 'O';
    }
    
    checkWinner() {
        const lines = [
            [0, 1, 2], [3, 4, 5], [6, 7, 8],
            [0, 3, 6], [1, 4, 7], [2, 5, 8],
            [0, 4, 8], [2, 4, 6]
        ];
        
        for (let [a, b, c] of lines) {
            if (this.board[a] && this.board[a] === this.board[b] && this.board[a] === this.board[c]) {
                return this.board[a];
            }
        }
        
        return this.board.every(cell => cell !== null) ? 'tie' : null;
    }
    
    render() {
        const gameContainer = document.getElementById('game');
        gameContainer.innerHTML = `
            <h2>Tic Tac Toe</h2>
            <p>Players: ${this.players.length}/2</p>
            ${this.winner ? 
                `<p>${this.winner === 'tie' ? "It's a tie!" : `Player ${this.winner} wins!`}</p>` :
                `<p>Current turn: ${this.currentPlayer}</p>`
            }
            <div class="board">
                ${this.board.map((cell, index) => 
                    `<button class="cell" onclick="game.makeMove(${index})" ${
                        !this.isMyTurn() || cell || this.winner ? 'disabled' : ''
                    }>${cell || ''}</button>`
                ).join('')}
            </div>
        `;
    }
}

// Initialize game
const game = new TicTacToe();
```

## Quick Setup Guide

1. **Choose a template** from the list above
2. **Copy the code** into your project
3. **Install dependencies**:
   ```bash
   npm install @PlayroomKit/client @PlayroomKit/react
   ```
4. **Get your API key** from the PlayroomKit dashboard
5. **Replace the API key** in the template code
6. **Run your game** and start playing!
