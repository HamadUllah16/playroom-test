# React Integration

PlayroomKit provides React hooks and components for seamless integration with React applications.

## Installation

```bash
npm install @PlayroomKit/react
```

## Setup

```jsx
import { PlayroomProvider } from '@PlayroomKit/react';

function App() {
    return (
        <PlayroomProvider apiKey="your-api-key">
            <Game />
        </PlayroomProvider>
    );
}
```

## Hooks

### usePlayroom

```jsx
import { usePlayroom } from '@PlayroomKit/react';

function GameComponent() {
    const { isConnected, players, myPlayer } = usePlayroom();
    
    if (!isConnected) {
        return <div>Connecting...</div>;
    }
    
    return (
        <div>
            <h2>Players in game: {players.length}</h2>
            <p>You are: {myPlayer.name}</p>
        </div>
    );
}
```

### usePlayroomState

```jsx
import { usePlayroomState } from '@PlayroomKit/react';

function ScoreBoard() {
    const [score, setScore] = usePlayroomState('score', { player1: 0, player2: 0 });
    
    return (
        <div>
            <h3>Score</h3>
            <p>Player 1: {score.player1}</p>
            <p>Player 2: {score.player2}</p>
            <button onClick={() => setScore({...score, player1: score.player1 + 1})}>
                Player 1 Score!
            </button>
        </div>
    );
}
```

### useRpc

```jsx
import { useRpc } from '@PlayroomKit/react';

function GameActions() {
    const sendAttack = useRpc('playerAttack');
    
    useRpc('playerAttack', (data, sender) => {
        console.log(`${sender.name} attacked!`);
    });
    
    return (
        <button onClick={() => sendAttack({ damage: 25 })}>
            Attack!
        </button>
    );
}
```

## Game Example

```jsx
import React from 'react';
import { usePlayroom, usePlayroomState, useRpc } from '@PlayroomKit/react';

function TicTacToe() {
    const { players, myPlayer } = usePlayroom();
    const [board, setBoard] = usePlayroomState('board', Array(9).fill(null));
    const [currentPlayer, setCurrentPlayer] = usePlayroomState('currentPlayer', 'X');
    
    const makeMove = useRpc('makeMove');
    
    useRpc('makeMove', (data) => {
        const newBoard = [...board];
        newBoard[data.position] = data.player;
        setBoard(newBoard);
        setCurrentPlayer(currentPlayer === 'X' ? 'O' : 'X');
    });
    
    const handleClick = (position) => {
        if (board[position] || !isMyTurn()) return;
        makeMove({ position, player: getMySymbol() });
    };
    
    const isMyTurn = () => {
        return (currentPlayer === 'X' && myPlayer.id === players[0]?.id) ||
               (currentPlayer === 'O' && myPlayer.id === players[1]?.id);
    };
    
    const getMySymbol = () => {
        return myPlayer.id === players[0]?.id ? 'X' : 'O';
    };
    
    return (
        <div className="tic-tac-toe">
            <h2>Tic Tac Toe</h2>
            <p>Current turn: {currentPlayer}</p>
            <div className="board">
                {board.map((cell, index) => (
                    <button
                        key={index}
                        className="cell"
                        onClick={() => handleClick(index)}
                        disabled={!isMyTurn() || cell !== null}
                    >
                        {cell}
                    </button>
                ))}
            </div>
        </div>
    );
}
```
